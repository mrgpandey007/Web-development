const express = require("express"); const mysql = require("mysql"); const bodyParser = require("body-parser"); const path = require("path"); const app = express(); app.use(bodyParser.urlencoded({ extended: true })); app.use(express.static(".")); // serves p7.html directly // Serve p7.html at root app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "p7.html")); }); // MySQL connection const db = mysql.createConnection({ host: "localhost", user: "fsduser", // or fsduser if you created one password: "fsdpassword", // update with your actual password database: "students_db" }); db.connect(err => { if (err) { console.error("MySQL connection error:", err); } else { console.log("Connected to MySQL database 'students_db'"); } }); // Route: Add student app.post("/add", (req, res) => { const { name, email } = req.body; db.query("CALL addStudent(?, ?)", [name, email], (err) => { if (err) { console.error("/add error:", err); res.status(500).send("Error inserting student: " + err.message); return; } res.send("Student added successfully"); }); }); // Route: Get students app.get("/students", (req, res) => { db.query("CALL getStudents()", (err, results) => { if (err) { console.error("/students error:", err); res.status(500).send("Error fetching students: " + err.message); return; } res.json(results && results[0] ? results[0] : []); }); }); app.listen(3000, () => console.log("http://localhost:3000"));
